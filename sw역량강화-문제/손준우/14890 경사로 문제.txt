#include<iostream>
#include<algorithm>
using namespace std;
int N,L;
int a[101][101];
int b[101][101];
int ans=0;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    cin >> N >> L;
    for(int i=1 ; i<=N ; i++){
        for(int j=1 ; j<=N ; j++){
            cin >> a[i][j];
            b[j][i] = a[i][j];
        }
    }
    int temp=a[i][1];
    for(int i=1 ; i<=N ; i++){
        for (int j = 1; j <= N; j++){
            temp = a[i][j];
            if(temp == a[i][j+1]){
            	j++;
                continue;
            }else if(temp != a[i][j+1]-1 && temp != a[i][j+1]+1){
                break;
            } else if (a[i][j] < temp){
            	j++;
                if(L==1){
                    continue;
                }
                int k;
                for(k=0 ; k<L ; k++){
                	cout << j << j+k << ' ';
                    if(a[i][j]==a[i][j+k]){
                        continue;
                    }else{
                        break;
                    }
                }
                if(k!=L){
                    break;
                }else{
                    j=j+L-1;
                    continue;
                }
            }  
            //cout << "end" << j << ' ' << N <<'\n';
        }
       
        if(j==N){
        	cout << "k"<< i << '\n';
            ans++;
        }else {
        	cout << "kn" << i << '\n';
        }
    }
    for(int i=1 ; i<=N ; i++){
        int j=1;
        while(j<N){
        	//cout << "start" << j << ' ' << N <<'\n';
            temp = a[j][i];
            //cout << "start2" << j << ' ' << N <<'\n';
            if(temp == a[j+1][i]){
            	//cout << temp << ' ' <<  a[j][i] << ' ' << j << '\n';
                j++;
                continue;
            }else if(temp != a[j+1][i]-1 && temp != a[j+1][i]+1){
                break;
            }else{
            	j++;
                if(L==1){
                    continue;
                }
                int k;
                for(k=1 ; k<L ; k++){
                    if(a[i][j]==a[j+k][i]){
                        continue;
                    }else{
                        break;
                    }
                }
                if(k!=L){
                    break;
                }else{
                    j=j+L-1;
                    continue;
                }
            }               
        }
        //cout << j << N <<"@@@@@@@@" <<'\n';
        if(j==N){
        	cout << i << '\n';
            ans++;
        }
    }
    cout << ans;
}
